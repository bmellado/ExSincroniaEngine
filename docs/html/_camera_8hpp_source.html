153 // Calculates the front vector from the Camera's (updated) Eular Angles
154 void updateCameraVectors()
155 {
156 // Calculate the new Front vector
157 glm::vec3 front;
158 front.x = cos(glm::radians(this->yaw)) * cos(glm::radians(this->pitch));
159 front.y = sin(glm::radians(this->pitch));
160 front.z = sin(glm::radians(this->yaw)) * cos(glm::radians(this->pitch));
161 this->front = glm::normalize(front);
162 // Also re-calculate the Right and Up vector
163 this->right = glm::normalize(glm::cross(this->front, this->worldUp)); // Normalize the vectors, because their length gets closer to 0 the more you look up or down which results in slower movement.
