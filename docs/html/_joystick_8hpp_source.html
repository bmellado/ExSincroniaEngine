85 void pollJoysticks(JoystickContainer& joysticks)
86 {
87 for (int joystickId = GLFW_JOYSTICK_1; joystickId < GLFW_JOYSTICK_LAST; ++joystickId)
88 {
89 int const joystickConnected = glfwJoystickPresent(joystickId);
90 if (joystickConnected == GLFW_FALSE)
91 {
92 if (joysticks.count(joystickId) != 0)
93 {
94 /* joystickId has been disconnected.*/
95 joysticks.erase(joystickId);
96 }
97 continue;
98 }
99
100 // At this point we have a joystick with this id.
101 if (joysticks.count(joystickId) == 0)
102 {
103 /* joystickId has been connected.*/
104
105 int buttonsCount;
106 glfwGetJoystickButtons(joystickId, &buttonsCount);
107
108 int axesCount;
109 glfwGetJoystickAxes(joystickId, &axesCount);
110
111 joysticks.emplace(joystickId, Joystick(buttonsCount, axesCount));
112 }
113
114 Joystick& joystick = joysticks.at(joystickId);
115
116 int buttonsCount;
117 const unsigned char* buttons = glfwGetJoystickButtons(joystickId, &buttonsCount);
118 assert(joystick.buttons.size() == buttonsCount);
119
120 for (int buttonId = 0; buttonId < buttonsCount; ++buttonId)
121 {
122 joystick.buttons[buttonId] = buttons[buttonId] == GLFW_PRESS;
123 }
124
125 int axesCount;
126 const float* axes = glfwGetJoystickAxes(joystickId, &axesCount);
127 assert(joystick.axes.size() == axesCount);
128
129 for (int axesId = 0; axesId < axesCount; ++axesId)
130 {
131 joystick.axes[axesId] = axes[axesId];
132 }
