73 // Processes a node in a recursive fashion. Processes each individual mesh located at the node and repeats this process on its children nodes (if any).
74 void processNode(aiNode* node, const aiScene* scene)
75 {
76 // Process each mesh located at the current node
77 for (GLuint i = 0; i < node->mNumMeshes; i++)
78 {
79 // The node object only contains indices to index the actual objects in the scene.
80 // The scene contains all the data, node is just to keep stuff organized (like relations between nodes).
81 aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
82
83 this->meshes.push_back(this->processMesh(mesh, scene));
84 }
85
86 // After we've processed all of the meshes (if any) we then recursively process each of the children nodes
87 for (GLuint i = 0; i < node->mNumChildren; i++)
88 {
89 this->processNode(node->mChildren[i], scene);
90 }

93 Mesh processMesh(aiMesh* mesh, const aiScene* scene)
94 {
95 // Data to fill
96 vector<Vertex> vertices;
97 vector<GLuint> indices;
98 vector<Texture> textures;
99
100 // Walk through each of the mesh's vertices
101 for (GLuint i = 0; i < mesh->mNumVertices; i++)
102 {
103 Vertex vertex;
104 glm::vec3 vector; // We declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
105
106 // Positions
107 vector.x = mesh->mVertices[i].x;
108 vector.y = mesh->mVertices[i].y;
109 vector.z = mesh->mVertices[i].z;
110 vertex.Position = vector;
111
112 // Normals
113 vector.x = mesh->mNormals[i].x;
114 vector.y = mesh->mNormals[i].y;
115 vector.z = mesh->mNormals[i].z;
116 vertex.Normal = vector;
117
118 // Texture Coordinates
119 if (mesh->mTextureCoords[0]) // Does the mesh contain texture coordinates?
120 {
121 glm::vec2 vec;
122 // A vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
123 // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
124 vec.x = mesh->mTextureCoords[0][i].x;
125 vec.y = mesh->mTextureCoords[0][i].y;
126 vertex.TexCoords = vec;
127 }
128 else
129 {
130 vertex.TexCoords = glm::vec2(0.0f, 0.0f);
131 }
132
133 vertices.push_back(vertex);
134 }
135
136 // Now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
137 for (GLuint i = 0; i < mesh->mNumFaces; i++)
138 {
139 aiFace face = mesh->mFaces[i];
140 // Retrieve all indices of the face and store them in the indices vector
141 for (GLuint j = 0; j < face.mNumIndices; j++)
142 {
143 indices.push_back(face.mIndices[j]);
144 }
145 }
146
147 // Process materials
148 if (mesh->mMaterialIndex >= 0)
149 {
150 aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
151 // We assume a convention for sampler names in the shaders. Each diffuse texture should be named
152 // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
153 // Same applies to other texture as the following list summarizes:
154 // Diffuse: texture_diffuseN
155 // Specular: texture_specularN
156 // Normal: texture_normalN
157
158 // 1. Diffuse maps
159 vector<Texture> diffuseMaps = this->loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
160 textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
161
162 // 2. Specular maps
163 vector<Texture> specularMaps = this->loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
164 textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
165 }
166
167 // Return a mesh object created from the extracted mesh data
168 return Mesh(vertices, indices, textures);
169 }

171 // Checks all material textures of a given type and loads the textures if they're not loaded yet.
172 // The required info is returned as a Texture struct.
173 vector<Texture> loadMaterialTextures(aiMaterial* mat, aiTextureType type, string typeName)
174 {
175 vector<Texture> textures;
176
177 for (GLuint i = 0; i < mat->GetTextureCount(type); i++)
178 {
179 aiString str;
180 mat->GetTexture(type, i, &str);
181
182 // Check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
183 GLboolean skip = false;
184
185 for (GLuint j = 0; j < textures_loaded.size(); j++)
186 {
187 if (textures_loaded[j].path == str)
188 {
189 textures.push_back(textures_loaded[j]);
190 skip = true; // A texture with the same filepath has already been loaded, continue to next one. (optimization)
191
192 break;
193 }
194 }
195
196 if (!skip)
197 { // If texture hasn't been loaded already, load it
198 Texture texture;
199 texture.id = TextureFromFile(str.C_Str(), this->directory);
200 texture.type = typeName;
201 texture.path = str;
202 textures.push_back(texture);
203
204 this->textures_loaded.push_back(texture); // Store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
205 }
206 }
207
208 return textures;
209 }

213{

227 glGenerateMipmap(GL_TEXTURE_2D);
