81template <typename PipelineType>
82void drawSceneGraphNode(
83 SceneGraphNodePtr nodePtr,
84 const PipelineType& pipeline,
85 const std::string& transformName,
86 const Matrix4f& parentTransform = Transformations::identity())
87{
88 // Composing the transformations through this path
89 Matrix4f newTransform = parentTransform * nodePtr->transform;
90
91 if (nodePtr->gpuShapeMaybe.has_value())
92 {
93 auto const shapePtr = nodePtr->gpuShapeMaybe.value();
94 auto const& shape = *shapePtr;
95 glUniformMatrix4fv(glGetUniformLocation(pipeline.shaderProgram, transformName.c_str()), 1, GL_FALSE, newTransform.data());
96 pipeline.drawCall(shape);
97 }
98
99 // If the child node is not a leaf, it MUST be a SceneGraphNode,
100 // so this draw function is called recursively
101 for (auto childPtr : nodePtr->childs)
102 drawSceneGraphNode(childPtr, pipeline, transformName, newTransform);
103}
